package seleniumTest;

public class Debugging {

	public static void main(String[] args) {
		int e=17, f=20, g=e+f;
		
		System.out.println(g);
		

		int a=20, b=45, c=a+b;
		System.out.println(c);
		a=++a;
		b=++b;
		c=a+b;
		if(c==67) {
			System.out.println("true");
		}else {
			System.out.println("false");
		}
		
		System.out.println(c);
		
		
	}
}
